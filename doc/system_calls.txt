int sys_create_thread(void* start_point, uint priority, void* arg = NULL);
    returns:
    0 - no enough memory
    non 0 - ok, returned value is new thread id

NO_RETURN sys_exit_thread();

void sys_write();
    write a character stored in an al register to stdout

void sys_read();
    read a character from stdin and store it in al

int sys_is_thread_running(int thread_id)
    returns:
    0 - not running
    1 - running
